# This is the cmake config file for the OnlineQueries plugin
SET(ONLINEQUERIES_MAJOR 0)
SET(ONLINEQUERIES_MINOR 0)
SET(ONLINEQUERIES_PATCH 2)
SET(ONLINEQUERIES_VERSION "${ONLINEQUERIES_MAJOR}.${ONLINEQUERIES_MINOR}.${ONLINEQUERIES_PATCH}")
ADD_DEFINITIONS(
-DONLINEQUERIES_MAJOR=${ONLINEQUERIES_MAJOR}
-DONLINEQUERIES_MINOR=${ONLINEQUERIES_MINOR}
-DONLINEQUERIES_PATCH=${ONLINEQUERIES_PATCH}
-DONLINEQUERIES_PLUGIN_VERSION="${ONLINEQUERIES_VERSION}"
)

ADD_DEFINITIONS(-DONLINEQUERIES_PLUGIN_LICENSE="GNU GPLv2 or later")

ADD_SUBDIRECTORY( src )

SET(CMAKE_INSTALL_PREFIX $HDATALOC)
INSTALL(FILES DESTINATION "modules/OnlineQueries")
